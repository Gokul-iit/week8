import pandas as pd
import matplotlib.pyplot as plt
import textwrap
import os
from io import StringIO, BytesIO
import base64

# Sample CSV (first 5 rows as provided by user)
csv_data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Operations,Europe,71.29,9,3.9
EMP002,HR,North America,83.87,5,4.6
EMP003,IT,Africa,63.54,11,4.3
EMP004,Finance,Asia Pacific,73.59,15,4.6
EMP005,IT,Latin America,81.32,10,3.6
"""

# Load into DataFrame
df = pd.read_csv(StringIO(csv_data))

# Calculate frequency count for the "IT" department
it_count = df[df['department'] == 'IT'].shape[0]
total_count = df.shape[0]
it_proportion = it_count / total_count if total_count else 0

# Print the frequency count to the console
print(f"IT department frequency (count): {it_count}")
print(f"IT department proportion: {it_proportion:.2f} ({it_count}/{total_count})")

# Create a histogram showing distribution of departments
dept_counts = df['department'].value_counts().sort_index()

fig, ax = plt.subplots(figsize=(8,5))
dept_counts.plot(kind='bar', ax=ax)
ax.set_title("Distribution of Departments")
ax.set_xlabel("Department")
ax.set_ylabel("Count")
plt.tight_layout()
